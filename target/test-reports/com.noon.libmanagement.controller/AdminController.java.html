package com.noon.libmanagement.controller;

import com.noon.libmanagement.dto.BookDto;
import com.noon.libmanagement.dto.BookDtoWithBookingId;
import com.noon.libmanagement.dto.RemoveBookDto;
import com.noon.libmanagement.exceptions.InvalidArgumentException;
import com.noon.libmanagement.exceptions.ServiceException;
import com.noon.libmanagement.response.ResponseDto;
import com.noon.libmanagement.service.AdminService;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.PutMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;
import com.noon.libmanagement.exceptions.IllegalAccessException;

@RestController
@RequiredArgsConstructor
public class AdminController {
	private final AdminService adminService;

	@PostMapping(path = "/api/v1/addBook")
	public ResponseDto<Boolean> addBook(@RequestBody final BookDto bookDto)
			throws InvalidArgumentException, IllegalAccessException {
		return new ResponseDto<>(adminService.addBook(bookDto, bookDto.getAdminId()));
	}

	@PutMapping(path = "/api/v1/addBook")
	public ResponseDto<Boolean> addBook(@RequestParam final BookDtoWithBookingId bookDtoWithBookingId)
			throws InvalidArgumentException, IllegalAccessException {
		return new ResponseDto<>(adminService
				.addBook(bookDtoWithBookingId.getBookingId(), bookDtoWithBookingId.getAdminId(),
						bookDtoWithBookingId.getNumberOfPieces()));
	}

	@PutMapping(path = "/api/v1/removeBook")
	public ResponseDto<Boolean> addBook(@RequestParam RemoveBookDto removeBookDto)
			throws ServiceException {
		return new ResponseDto<>(adminService.removeBook(removeBookDto.getBookId(), removeBookDto.getAdminId()));
	}
}
