package com.noon.libmanagement.mapper;

import com.noon.libmanagement.dto.BookDto;
import com.noon.libmanagement.exceptions.InvalidArgumentException;
import com.noon.libmanagement.models.Book;
import org.springframework.stereotype.Component;

@Component
public class BookMapper {
	public Book getEntityFromDto(final BookDto bookDto) throws InvalidArgumentException {
		if (bookDto == null)
			throw new InvalidArgumentException("source can not be null");
		return Book.builder().bookType(bookDto.getBookType()).authorName(bookDto.getAuthorName())
				.bookName(bookDto.getBookName()).count(
						bookDto.getCount()).build();
	}
}
