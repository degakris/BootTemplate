package com.noon.libmanagement.controller;

import com.noon.libmanagement.dto.BorrowBookRequestDto;
import com.noon.libmanagement.dto.BorrowBookWithAuthorName;
import com.noon.libmanagement.dto.ReturnBookRequestDto;
import com.noon.libmanagement.dto.ReturnBookRequestWithAuthorDto;
import com.noon.libmanagement.models.Booking;
import com.noon.libmanagement.response.ResponseDto;
import com.noon.libmanagement.service.BookIssueService;
import lombok.RequiredArgsConstructor;
import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RestController;
import com.noon.libmanagement.exceptions.IllegalAccessException;

@RestController
@RequiredArgsConstructor
public class LibManagementController {

	private final BookIssueService bookIssueService;

	@PostMapping(path = "api/v1/borrowBook")
	public ResponseDto<Booking> borrowBook(@RequestBody final BorrowBookRequestDto borrowBookRequestDto)
			throws IllegalAccessException {
		return new ResponseDto<>(bookIssueService
				.borrowBook(borrowBookRequestDto.getUserId(), borrowBookRequestDto.getBookId(),
						borrowBookRequestDto.getDays()));
	}

	@PostMapping(path = "api/v1/borrowBookWithAuthor")
	public ResponseDto<Booking> borrowBook(@RequestBody final BorrowBookWithAuthorName bookWithAuthorName)
			throws IllegalAccessException {
		return new ResponseDto<>(bookIssueService
				.borrowBook(bookWithAuthorName.getUserId(), bookWithAuthorName.getAuthorName(),
						bookWithAuthorName.getBookName(), bookWithAuthorName.getDays()));
	}

	@PostMapping(path = "api/v1/returnBook")
	public ResponseDto<Boolean> returnBook(@RequestBody final ReturnBookRequestDto returnBookRequestDto)
			throws IllegalAccessException {
		return new ResponseDto<>(
				bookIssueService.returnBook(returnBookRequestDto.getUserId(), returnBookRequestDto.getBookId()));
	}

	@PostMapping(path = "api/v1/returnBookWithBookingId")
	public ResponseDto<Boolean> returnBook(@RequestBody final Integer bookingId)
			throws IllegalAccessException {
		return new ResponseDto<>(bookIssueService.returnBook(bookingId));
	}

	@PostMapping(path = "api/v1/returnBookWithAuthor")
	public ResponseDto<Boolean> returnBook(
			@RequestBody final ReturnBookRequestWithAuthorDto returnBookRequestWithAuthorDto)
			throws IllegalAccessException {
		return new ResponseDto<>(bookIssueService
				.returnBook(returnBookRequestWithAuthorDto.getUserId(), returnBookRequestWithAuthorDto.getAuthorName(),
						returnBookRequestWithAuthorDto.getBookName()));
	}

}
